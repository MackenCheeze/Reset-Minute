let ambientCtx, gainNode, oscA, oscB, sub;
function startAmbient(){ambientCtx=new(window.AudioContext||window.webkitAudioContext)();gainNode=ambientCtx.createGain();gainNode.gain.setValueAtTime(0.00001,ambientCtx.currentTime);gainNode.connect(ambientCtx.destination);oscA=ambientCtx.createOscillator();oscA.type="sine";oscA.frequency.value=55;oscB=ambientCtx.createOscillator();oscB.type="sine";oscB.frequency.value=58;sub=ambientCtx.createOscillator();sub.type="sine";sub.frequency.value=28;oscA.connect(gainNode);oscB.connect(gainNode);sub.connect(gainNode);oscA.start();oscB.start();sub.start();gainNode.gain.exponentialRampToValueAtTime(0.12,ambientCtx.currentTime+4);}
function stopAmbient(){if(gainNode){gainNode.gain.exponentialRampToValueAtTime(0.00001,ambientCtx.currentTime+2);}if(oscA)oscA.stop(ambientCtx.currentTime+2.2);if(oscB)oscB.stop(ambientCtx.currentTime+2.2);if(sub)sub.stop(ambientCtx.currentTime+2.2);}

/* BEEP */
function playBeep(){let ctx=new(window.AudioContext||window.webkitAudioContext)();let osc=ctx.createOscillator();let gain=ctx.createGain();osc.type="sine";osc.frequency.setValueAtTime(432,ctx.currentTime);osc.connect(gain);gain.connect(ctx.destination);gain.gain.setValueAtTime(0.00001,ctx.currentTime);gain.gain.exponentialRampToValueAtTime(0.28,ctx.currentTime+0.18);gain.gain.exponentialRampToValueAtTime(0.00001,ctx.currentTime+1.1);osc.start();osc.stop(ctx.currentTime+1.1);}

/* RESET */
function beginReset(){

ring.style.opacity = 0.6;

let totalTime=60;
let startTime=Date.now();

let radius=progressCircle.r.baseVal.value;
let circumference=2*Math.PI*radius;
progressCircle.style.strokeDasharray=circumference;

function updateProgress(){
let elapsed=(Date.now()-startTime)/1000;
let offset=circumference-(elapsed/totalTime)*circumference;
progressCircle.style.strokeDashoffset=offset;
let remaining = Math.ceil(totalTime - elapsed);
ringTimer.innerText = remaining > 0 ? remaining : 0;
}

const canvas = document.getElementById('blobCanvas');
const ctx = canvas.getContext('2d');
canvas.width = 300;
canvas.height = 300;

function getTime(){return (Date.now()-startTime)/1000;}

const total = inhale + hold + exhale;

function ease(x){return x < 0.5 ? 2*x*x : 1 - Math.pow(-2*x+2,2)/2;}

function breath(t){
t %= total;
if(t < inhale) return ease(t / inhale);
if(t < inhale + hold) return 1;
return 1 - ease((t - inhale - hold) / exhale);
}

function getPhase(t){
t%=total;
if(t<inhale) return "Inhale";
if(t<inhale+hold) return "Hold";
return "Exhale";
}

function noise(a,t,s=1){
return Math.sin(a*2+t)*6*s +
Math.cos(a*3-t*0.7)*5*s +
Math.sin(a*6+t*0.3)*3*s;
}

function blob(cx,cy,r,pts,t,s=1){
ctx.beginPath();
for(let i=0;i<=pts;i++){
let a=(i/pts)*Math.PI*2;
let rad=r+noise(a,t,s);
let x=cx+Math.cos(a)*rad;
let y=cy+Math.sin(a)*rad;
i?ctx.lineTo(x,y):ctx.moveTo(x,y);
}
ctx.closePath();
}

const lines=[];
for(let i=0;i<6;i++){
lines.push({offset:Math.random()*10,speed:0.5+Math.random()*0.5,thickness:0.5+Math.random()*1.5,radius:80+i*5});
}

function drawLines(cx,cy,scale,t){
lines.forEach((l,i)=>{
ctx.save();
ctx.translate(cx,cy);
ctx.rotate(t*0.05*l.speed+i);
ctx.beginPath();
for(let j=0;j<=160;j++){
let a=(j/160)*Math.PI*2;
let r=(l.radius*scale)+noise(a,t+l.offset,0.4);
let x=Math.cos(a)*r;
let y=Math.sin(a)*r;
j?ctx.lineTo(x,y):ctx.moveTo(x,y);
}
ctx.strokeStyle='rgba(180,200,255,0.15)';
ctx.lineWidth=l.thickness;
ctx.stroke();
ctx.restore();
});
}

function draw(){
let t = getTime();
ctx.clearRect(0,0,300,300);
let cx=150,cy=150;
let scale=0.5+breath(t)*0.6;

text.innerText = getPhase(t);

drawLines(cx,cy,scale,t);

for(let i=0;i<3;i++){
ctx.fillStyle='rgba(180,200,255,0.2)';
blob(cx,cy,70*(scale-i*0.08),120,t+i*0.4,1-i*0.2);
ctx.fill();
}

ctx.globalCompositeOperation='destination-out';
blob(cx,cy,35*scale,120,t+0.6,0.5);
ctx.fill();
ctx.globalCompositeOperation='lighter';

drawLines(cx,cy,scale*0.95,t);

requestAnimationFrame(draw);
}

draw();

resetTimeout=setInterval(()=>{
updateProgress();
let elapsed=(Date.now()-startTime)/1000;
if(elapsed>=60){
clearInterval(resetTimeout);
document.querySelector(".menu-btn").style.display="block";
stopAmbient();
ring.style.opacity=0;
document.getElementById("reset").classList.add("hidden");
document.getElementById("focus").classList.remove("hidden");
}
},50);

}
